## Match

在学习Rust后发现其内部的`match`方法的分支判断方式相当不错，仿照着这思路用TS做了个简单封装来实现类似的功能。为了足够简单没来支持Array类型的checks。
```ts
/**
 * 类似于Rust中的match功能实现
 * @example
 *   let message = 'hello'
 *   let response = match(message, {
 *     hello: (v) => 'the value is ${v}',
 *     goodbye () => 'hello to you too!',
 *     _ => (v) => 'something else ${v}'
 *   ])
 *   console.log(response) // prints 'the value is hello'
 *
 *   // no default and no match
 *   let number = '26'
 *   let res = match(number, {
 *     5: () => 'the value is hi',
 *     'test word': () => 'the value is test word',
 *   })
 *   console.log(response) // prints undefined
 */
export type MatchPattern<T extends string | number, R> = {
  [K in T]?: (value: K) => R
} & {
  _?: (value: T) => R // 默认处理器是可选的
}

// 定义并实现 match 函数
export default function match<T extends string | number, R>(
  value: T,
  checks: MatchPattern<T, R>
): R | undefined {
  if (checks[value]) {
    // @ts-expect-error 不做校验
    return checks[value](value)
  } else if (checks._) {
    return checks._(value)
  }

  // 如果没有匹配项且没有默认处理器，则返回 undefined
  return undefined
}

```
