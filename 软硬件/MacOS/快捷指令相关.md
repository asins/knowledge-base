## 在Finder中用Alacritty打开文件

- 新建快捷指令后，在快捷指令详细信息中勾选`作为快捷操作使用-访达`。
- 在操作库中加入`为Mac自动化运行JavaScript`，

```js
function run(input, parameters) {
    var app = Application.currentApplication();
    app.includeStandardAdditions = true;

    for (var i = 0; i < input.length; i++) {
        var itemPath = Path(input[i]);
        try {
            quotedString = itemPath.toString().replace(/([\\"])/g, '\\$1');
            // app.displayDialog("/Applications/Alacritty.app/Contents/MacOS/alacritty --working-directory '" + quotedString + "' &> /dev/null &")
            // 在后台打开Alacritty，使用&使命令在后台执行
            app.doShellScript("/Applications/Alacritty.app/Contents/MacOS/alacritty --working-directory '" + quotedString + "' &> /dev/null &");
            return
        } catch (error) {
            app.displayDialog("Alacritty可能没有安装");
            return;
        }
    }
}
```



```js

function run(input, parameters) {
    var app = Application.currentApplication();
    app.includeStandardAdditions = true;

    // 引号包裹字符串的函数，用于处理文件路径
    var quote = str => `"${str.replace(/([\\"])/g, '\\$1')}"`;

    // 错误信息收集
    var errors = [];

    // 使用for循环代替forEach
    for (var i = 0; i < input.length; i++) {
        var filePath = input[i].toString();
        try {
            // 检测是否为文件，若是则使用其目录，否则直接使用filePath
            var isFileCommand = `test -f ${quote(filePath)} && echo true || echo false`;
            var directoryPath = app.doShellScript(isFileCommand) === 'true' ? app.doShellScript(`dirname ${quote(filePath)}`) : filePath;
            directoryPath = directoryPath.toString().replace(/([\\"])/g, '\\$1');

            
            // 在后台打开Alacritty，使用&使命令在后台执行
            app.doShellScript(`/Applications/Alacritty.app/Contents/MacOS/alacritty --working-directory '${directoryPath}' &> /dev/null &`);
            
        } catch (error) {
            // 收集发生的错误信息，不立即处理
            errors.push("Item " + (i+1) + ": " + error.message);
        }
    }

    // 如果有错误发生，在循环结束后统一显示一次
    if(errors.length) {
        app.displayDialog("在处理过程中，以下项目发生错误:\n" + errors.join('\n'), {buttons: ["OK"], defaultButton: "OK"});
    }
}
```
