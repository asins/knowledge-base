```bash
# 对硬盘进行格式化
parted /dev/sdc
# 将磁盘格式转换为GPT（旧的MBR格式硬盘不支持2T以上硬盘，需要转换为GPT格式才支持）
(parted) mklabel gpt
# 将整块硬盘分一个区，分区使用使用xfs格式
(parted) mkpart primary xfs 0 -1
# 输入p查看配置情况
(parted) p
# 退出parted命令工具行
(parted) quite

# 开始格式化
mkfs.xfs /dev/sdc

# 设置开机自动挂载硬盘 
vim /etc/fstab
# 用blkid查看硬盘的UUID
blkid /dev/sdb
# 加入一行
UUID=xxxx /mnt/3T xfs defaults 1 2


```
